```sh
git submodule update --remote

# 初始化 + 更新 (首次)
git submodule update --init --recursive --remote

# 添加子模块
git submodule add https://github.com/stephane/libmodbus vendor/libmodbus

# 清理 libmodbus 源目录的配置痕迹，然后重建
cd vendor/libmodbus
make distclean || rm -f configure config.status config.log Makefile
cd - 

# Git 中用于清理工作目录中未跟踪文件的命令，它会删除那些 Git 没有追踪的文件和目录。
# 彻底清理，回到"纯净"状态（类似刚 clone 下来）
# 子模块狂喜
git clean -fdx 
```
